#!/usr/bin/env python3
"""
COMPLETE SIGN LANGUAGE TRAINING & RECOGNITION GUIDE

Step-by-step workflow:
"""

print("""
╔══════════════════════════════════════════════════════════════════════════╗
║          SIGN LANGUAGE RECOGNITION - COMPLETE WORKFLOW                  ║
╚══════════════════════════════════════════════════════════════════════════╝

THREE SIMPLE STEPS:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 1: COLLECT TRAINING DATA
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Command:
    python collect_data.py

  What happens:
    • Shows you each sign one at a time
    • Displays YOUR hand landmarks in real-time (green skeleton)
    • You perform the sign
    • Press SPACE to record 45 frames
    • Press ESC to skip a sign
    • Repeats until 50 samples per sign collected

  How to customize WHICH SIGNS:
    Edit collect_data.py, line ~127:
    
      SIGNS = [
          "hello",
          "thank_you",
          "yes",
          "no",
          ...
      ]
    
    Just edit the list to add/remove signs!

  Tips:
    • Good lighting helps hand detection
    • Move your hands clearly but not too fast
    • Different people, angles, speeds = better model

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: TRAIN THE MODEL
━━━━━━━━━━━━━━━━━━━━━━━

  Command:
    python train.py

  What happens:
    • Loads all your collected .npy files
    • Splits: 80% training, 20% testing
    • Trains a bidirectional GRU neural network
    • Shows progress each epoch:
      - Loss: how well model fits data
          - Accuracy: % correct predictions
          - Best! = better model found, saves it
        • Takes ~5-10 minutes on GPU/MPS

  How it works:
    1. Your hand landmark sequences go into the model
    2. Model learns patterns specific to each sign
    3. Tests on unseen data (20%)
    4. Keeps the best version (lowest test loss)

  Customize training:
    Edit train.py, top of file:
      BATCH_SIZE = 32    (higher = faster, needs more RAM)
      EPOCHS = 50        (more = longer training, better model)
      LEARNING_RATE = 0.001  (lower = more stable)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: RUN REAL-TIME RECOGNITION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Command:
    python -m realtime.realtime_inference

    conda activate sign_stream && streamlit run minimal_ui.py

  Or use:
    python run.py

  What happens:
    • Activates your webcam
    • Shows live hand landmarks
    • Every 45 frames, predicts the sign
    • Uses text-to-speech to say the sign name
    • Press ESC to quit

  How it works:
    1. MediaPipe detects your hands (green skeleton)
    2. Extracts hand features (positions, distances, angles, motion)
    3. Buffers 45 frames
    4. Feeds to trained model
    5. Model predicts which sign you're making
    6. If confident enough, speaks it out loud

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

FOLDER STRUCTURE (AUTO-CREATED):

  dataset/
    ├── hello/
    │   ├── 0.npy (50 samples)
    │   ├── 1.npy
    │   └── ...
    ├── thank_you/
    ├── yes/
    └── ... (one folder per sign)

  models/
    └── sign_model.pth (your trained model)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

TROUBLESHOOTING:

  "No data found" when training?
    → Run collect_data.py first!

  Hand landmarks not showing?
    → Check lighting
    → Make sure hand is visible in frame
    → Move closer to camera

  Poor recognition accuracy?
    → Collect more samples (increase SAMPLES_PER_SIGN)
    → Train for more epochs
    → Be more consistent with your signs

  Model not saving?
    → Check models/ folder exists and is writable

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

TIPS FOR BEST RESULTS:

  1. Consistent signing
     → Use same hand position for each sign
     → Same speed each time
     → Clear hand shapes

  2. Vary conditions
     → Different lighting
     → Different distances from camera
     → Different hand positions
     → This makes model more robust!

  3. More data = better model
     → 50 samples is minimum
     → 100+ samples much better
     → Different people = even better

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Ready? Start with:
  python collect_data.py

""")
